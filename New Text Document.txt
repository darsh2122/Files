public static void validateColumn1(Sheet sheet, String validValue) {
    for (int i = 1; i < sheet.getPhysicalNumberOfRows(); i++) {
        Row row = sheet.getRow(i);
        Cell cell = row.getCell(0);
        if (cell.getCellType() == CellType.STRING) {
            String value = cell.getStringCellValue();
            if (value.equals(validValue)) {
                System.out.println("Valid format at row " + row.getRowNum() + ", column 1");
            } else {
                System.out.println("Invalid format at row " + row.getRowNum() + ", column 1");
            }
        } else {
            System.out.println("Invalid cell type or format at row " + row.getRowNum() + ", column 1");
        }
    }
}
