import java.sql.*;
import java.util.List;

public class MyDatabaseChecker {
    private static final String JDBC_DRIVER = "com.microsoft.sqlserver.jdbc.SQLServerDriver";
    private static final String DB_URL = "jdbc:sqlserver://localhost;databaseName=myDatabase";
    private static final String USER = "username";
    private static final String PASSWORD = "password";

    public static boolean areValuesPresentInTable(List<String> values) throws SQLException {
        Connection conn = null;
        PreparedStatement stmt = null;
        ResultSet rs = null;

        try {
            Class.forName(JDBC_DRIVER);
            conn = DriverManager.getConnection(DB_URL, USER, PASSWORD);

            // Use a prepared statement to avoid SQL injection attacks
            String sql = "SELECT COUNT(*) FROM myTable WHERE myColumn IN (?)";
            stmt = conn.prepareStatement(sql);

            // Set the parameter value using a comma-separated list of values
            String param = String.join(",", values);
            stmt.setString(1, param);

            rs = stmt.executeQuery();
            if (rs.next()) {
                int count = rs.getInt(1);
                return count == values.size();
            } else {
                return false;
            }
        } catch (ClassNotFoundException e) {
            // Handle driver loading error
        } finally {
            if (rs != null) {
                rs.close();
            }
            if (stmt != null) {
                stmt.close();
            }
            if (conn != null) {
                conn.close();
            }
        }
    }
}
