            screenshots.append(os.path.join(images_dir, filename))

def insert_screenshots_to_word(screenshots, output_file):
    document = Document()
    for screenshot in screenshots:
        document.add_picture(screenshot)
    document.save(output_file)
    print(f"Screenshots inserted into '{output_file}' successfully.")
