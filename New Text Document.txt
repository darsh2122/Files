public static double convertAmount(String amount) {
        double convertedAmount = 0.0;

        if (amount.endsWith("CR")) {
            String numericalValue = amount.substring(0, amount.length() - 2);
            convertedAmount = Double.parseDouble(numericalValue);
        } else if (amount.endsWith("DR")) {
            String numericalValue = amount.substring(0, amount.length() - 2);
            convertedAmount = -Double.parseDouble(numericalValue);
        }

        return convertedAmount;
    }
