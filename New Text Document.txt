import java.awt.Rectangle;
import java.awt.Robot;
import java.awt.Toolkit;
import java.awt.image.BufferedImage;
import java.io.File;
import javax.imageio.ImageIO;
import javax.swing.JFrame;
import javax.swing.SwingUtilities;

public class WindowScreenshot {

    public static void main(String[] args) {
        // Run the screenshot code on the event dispatch thread
        SwingUtilities.invokeLater(() -> {
            // Create a JFrame representing the target window
            JFrame targetWindow = new JFrame("Target Window");

            // Make the target window visible to capture its contents
            targetWindow.setVisible(true);

            // Capture the screenshot of the target window
            captureScreenshot(targetWindow);

            // Close the target window
            targetWindow.dispose();
        });
    }

    private static void captureScreenshot(JFrame window) {
        try {
            // Get the window bounds
            Rectangle windowBounds = window.getBounds();

            // Create a Robot object
            Robot robot = new Robot();

            // Capture the screenshot of the window using Robot's createScreenCapture method
            BufferedImage screenshotImage = robot.createScreenCapture(windowBounds);

            // Save the captured screenshot to a file
            String screenshotPath = "C:/path/to/save/screenshot.png";
            ImageIO.write(screenshotImage, "png", new File(screenshotPath));

            System.out.println("Screenshot captured and saved successfully.");
        } catch (Exception ex) {
            System.err.println("Failed to capture screenshot: " + ex.getMessage());
        }
    }
}
